import { useState } from 'react';
import { Button } from '../components/ui/button';
import { Input } from '../components/ui/input';
import { Label } from '../components/ui/label';
import { PicksAdmin } from './PicksAdmin';
import { FinalizePicks } from './FinalizePicks';

interface AdminProps {
  onBackToSite?: () => void;
}

type AdminPage = 'picks-admin' | 'finalize-picks';

export function Admin({ onBackToSite }: AdminProps) {
  const [password, setPassword] = useState('');
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  const [currentPage, setCurrentPage] = useState<AdminPage>('picks-admin');

  const handleSignIn = (e: React.FormEvent) => {
    e.preventDefault();
    // Authentication logic will go here when you connect a backend
    // For now, any password will work
    console.log('Sign in attempted with password:', password);
    setIsAuthenticated(true);
  };

  const handleSignOut = () => {
    setIsAuthenticated(false);
    setPassword('');
    setCurrentPage('picks-admin');
  };

  const handleNavigateToFinalize = () => {
    setCurrentPage('finalize-picks');
  };

  const handleBackToPicksAdmin = () => {
    setCurrentPage('picks-admin');
  };

  // Show appropriate admin page if authenticated
  if (isAuthenticated) {
    if (currentPage === 'finalize-picks') {
      return (
        <FinalizePicks 
          onSignOut={handleSignOut} 
          onBack={handleBackToPicksAdmin}
        />
      );
    }
    
    return (
      <PicksAdmin 
        onSignOut={handleSignOut} 
        onNavigateToFinalize={handleNavigateToFinalize}
      />
    );
  }

  return (
    <div className="min-h-screen bg-background flex flex-col">
      {/* Header */}
      <header className="border-b border-border">
        <div className="container mx-auto px-4 py-3">
          <div className="flex items-center">
            <img 
              src="/logo.png" 
              alt="Well Spent Style" 
              className="h-16 cursor-pointer"
              onClick={onBackToSite}
            />
          </div>
        </div>
      </header>

      {/* Main Content */}
      <main className="flex-1 flex items-center justify-center px-4 py-20">
        <div className="w-full max-w-md">
          <div className="border border-border p-12 bg-white">
            <h1 
              className="text-3xl mb-2 tracking-tight" 
              style={{ fontFamily: 'DM Sans, sans-serif', fontWeight: 700 }}
            >
              Admin Access
            </h1>
            <p 
              className="text-muted-foreground mb-10" 
              style={{ fontFamily: 'Crimson Pro, serif' }}
            >
              Enter your password to manage sales and content.
            </p>

            <form onSubmit={handleSignIn} className="space-y-6">
              <div className="space-y-2">
                <Label htmlFor="password">Password</Label>
                <Input
                  id="password"
                  type="password"
                  value={password}
                  onChange={(e) => setPassword(e.target.value)}
                  placeholder="Enter your password"
                  className="h-12"
                  required
                />
              </div>

              <Button 
                type="submit" 
                className="w-full h-12"
                style={{ fontFamily: 'DM Sans, sans-serif' }}
              >
                Sign In
              </Button>
            </form>
          </div>

          <div className="mt-6 text-center">
            <button 
              onClick={onBackToSite}
              className="text-sm text-muted-foreground hover:text-foreground transition-colors cursor-pointer"
              style={{ fontFamily: 'Crimson Pro, serif' }}
            >
              ‚Üê Back to site
            </button>
          </div>
        </div>
      </main>
    </div>
  );
}
