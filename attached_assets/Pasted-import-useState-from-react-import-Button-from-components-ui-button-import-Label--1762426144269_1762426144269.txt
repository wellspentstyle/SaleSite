import { useState } from 'react';
import { Button } from '../components/ui/button';
import { Label } from '../components/ui/label';
import { Textarea } from '../components/ui/textarea';
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from '../components/ui/select';
import { mockSales } from '../data/mockSales';

interface PicksAdminProps {
  onSignOut: () => void;
  onNavigateToFinalize: () => void;
}

export function PicksAdmin({ onSignOut, onNavigateToFinalize }: PicksAdminProps) {
  const [selectedSaleId, setSelectedSaleId] = useState<string>('');
  const [urls, setUrls] = useState<string>('');
  const [isLoading, setIsLoading] = useState(false);

  const handleScrapePicks = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (!selectedSaleId) {
      alert('Please select a sale');
      return;
    }

    if (!urls.trim()) {
      alert('Please enter at least one URL');
      return;
    }

    setIsLoading(true);

    // Parse URLs (one per line)
    const urlList = urls.split('\n').filter(url => url.trim() !== '');
    
    // Backend logic will go here to:
    // 1. Send URLs to scraping service
    // 2. Extract product details (image, title, price)
    // 3. Save picks to database associated with the selected sale
    console.log('Scraping picks for sale:', selectedSaleId);
    console.log('URLs to scrape:', urlList);

    // Simulate API call
    setTimeout(() => {
      setIsLoading(false);
      // Navigate to Finalize Picks page after successful scraping
      onNavigateToFinalize();
    }, 2000);
  };

  return (
    <div className="min-h-screen bg-background flex flex-col">
      {/* Header */}
      <header className="border-b border-border">
        <div className="container mx-auto px-4 py-3">
          <div className="flex items-center justify-between">
            <img 
              src="/logo.png" 
              alt="Well Spent Style" 
              className="h-16"
            />
            <button
              onClick={onSignOut}
              className="text-sm text-muted-foreground hover:text-foreground transition-colors"
              style={{ fontFamily: 'DM Sans, sans-serif' }}
            >
              Sign Out
            </button>
          </div>
        </div>
      </header>

      {/* Main Content */}
      <main className="flex-1 px-4 py-20">
        <div className="container mx-auto max-w-4xl">
          <div className="mb-12">
            <h1 
              className="text-4xl mb-3 tracking-tight" 
              style={{ fontFamily: 'DM Sans, sans-serif', fontWeight: 700 }}
            >
              Manage Picks
            </h1>
            <p 
              className="text-muted-foreground text-lg" 
              style={{ fontFamily: 'Crimson Pro, serif' }}
            >
              Upload product URLs to scrape and add curated picks to your sales.
            </p>
          </div>

          <div className="border border-border p-12 bg-white">
            <form onSubmit={handleScrapePicks} className="space-y-8">
              {/* Sale Selection */}
              <div className="space-y-3">
                <Label 
                  htmlFor="sale" 
                  className="text-xl"
                  style={{ fontFamily: 'DM Sans, sans-serif', fontWeight: 700 }}
                >
                  Select Sale
                </Label>
                <Select value={selectedSaleId} onValueChange={setSelectedSaleId}>
                  <SelectTrigger className="h-12 text-sm">
                    <SelectValue placeholder="Choose a sale to add picks to..." />
                  </SelectTrigger>
                  <SelectContent>
                    {mockSales.map((sale) => (
                      <SelectItem key={sale.id} value={sale.id}>
                        {sale.brand} - {sale.discount} Off
                        {sale.startDate && ` (${new Date(sale.startDate).toLocaleDateString()})`}
                      </SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>

              {/* URL Input */}
              <div className="space-y-3">
                <Label 
                  htmlFor="urls" 
                  className="text-xl"
                  style={{ fontFamily: 'DM Sans, sans-serif', fontWeight: 700 }}
                >
                  Product URLs
                </Label>
                <Textarea
                  id="urls"
                  value={urls}
                  onChange={(e) => setUrls(e.target.value)}
                  placeholder="Paste product URLs here, one per line:&#10;https://example.com/product-1&#10;https://example.com/product-2&#10;https://example.com/product-3"
                  className="min-h-[300px] text-sm"
                  style={{ fontFamily: 'monospace' }}
                  required
                />
              </div>

              {/* Submit Button */}
              <div className="flex gap-4 pt-4">
                <Button 
                  type="submit" 
                  className="h-12 px-8"
                  style={{ fontFamily: 'DM Sans, sans-serif' }}
                  disabled={isLoading}
                >
                  {isLoading ? 'Scraping...' : 'Scrape Picks'}
                </Button>
                <Button 
                  type="button"
                  variant="outline"
                  className="h-12 px-8"
                  style={{ fontFamily: 'DM Sans, sans-serif' }}
                  onClick={() => {
                    setUrls('');
                    setSelectedSaleId('');
                  }}
                  disabled={isLoading}
                >
                  Clear
                </Button>
              </div>
            </form>
          </div>
        </div>
      </main>
    </div>
  );
}
